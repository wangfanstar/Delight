§6-1 查找第一个0字节 — Find First 0-Byte
// THE STRLEN TRICK — SWAR：寄存器内的SIMD C 字符串以 0 字节（'\0'）终止，strlen 的快速实现必须 一次检测一整个 word 中是否有 0 字节，而不是逐字节循环。 Warren §6-1 给出的方法是经典的 SWAR 技术： 用一个巧妙的算术表达式 (x - 0x01010101) & ~x & 0x80808080， 在 4 条指令内同时检测 4 个字节是否为 0。 这比逐字节检测快 4 倍，且完全无分支，是 glibc、musl、Apple libc 等所有主流 C 库 strlen 实现的核心算法，可追溯至 1987 年的 Alan Mycroft。

§6-1 三种 zbytel 实现对比
// 朴素版 zbytel: 逐字节检测, 最坏 4 次比较 int zbytel_naive(unsigned x) { if ((x >> 24) == 0) return 0; else if ((x & 0x00FF0000) == 0) return 1; else if ((x & 0x0000FF00) == 0) return 2; else if ((x & 0x000000FF) == 0) return 3; else return 4; } // Warren Figure 6-1 // 指令数: 最多 5×(cmp+branch) = 10指令 // 缺点: 多分支, 无法流水线化 // 优点: 最直观, 无需理解位技巧
// Warren Figure 6-2: SWAR 无分支版 — 7条指令 // 原理: 一次检测4个字节是否有0 int zbytel(unsigned x) { unsigned y; // 原始字节: 00 80 other // x-0x0101: FF 7F varies // ~x: FF 7F varies // 两者AND: FF 00 0xxxxxxx // AND 0x8080: 80 00 0 (bit7清零) y = (x & 0x7F7F7F7F) + 0x7F7F7F7F; y = ~(y | x | 0x7F7F7F7F); // y 的每个字节: 0字节→0x80, 非0字节→0x00 int n = nlz(y) >> 3; // 前导零字节数 = 位置 return n; // 0~3: 位置; 4: 无0字节 } // 等价形式 (更常见于 glibc): int has_zero_byte(unsigned x) { return ((x - 0x01010101) & ~x & 0x80808080) != 0; } // 分析: // 对每字节 b: // b=0: 0-1=FF→bit7=1, ~0=FF→bit7=1 → AND→bit7=1 ✓ // b=0x80: 80-1=7F→bit7=0 → AND→bit7=0 ✓ // b≠0,≠80: 减法不会产生bit7借位 → bit7=0 ✓ // 7条指令(包含nlz), 完全无分支, 可流水线!
// §6-1 扩展: 搜索字节值在某范围内的首字节 // 应用: 找第一个大写字母、第一个空白字符等 // 找第一个字节值在 [lo, hi] 范围内的字节 // 方法: 利用带符号溢出技巧 int find_byte_in_range(unsigned x, unsigned char lo, unsigned char hi) { // 将 x 中每字节减去 lo, 检查是否 < (hi-lo+1) unsigned L = lo * 0x01010101; unsigned H = (hi - lo + 1) * 0x01010101; unsigned d = x - L; unsigned y = (d | (d - H)) & ~d & 0x80808080; return nlz(y) >> 3; } // 实例: 找第一个大写字母 [0x41, 0x5A] // find_byte_in_range(x, 'A', 'Z') // 64位机扩展: 一次检测8字节 // 将常数扩展为64位: 0x0101010101010101ULL 等 // glibc strlen 使用此技术, 每次加载一个64位字 // ARM64 也可用 SIMD: VMOV.8b + VCEQ.8b + VMOVMSK // Warren 指出: 此方法可适应任意字节范围 ≥128 // 即对范围0~137的搜索: 将 y = y | x 改为 y = y & x

🏭 strlen 快速实现 — 字对齐 + SWAR 结合
// 实际 strlen 实现框架 (以32位为例, 64位类似) // 参考 glibc / musl / BSD libc 实现策略 size_t fast_strlen(const char *s) { const char *p = s; // 阶段1: 逐字节处理直到字对齐 (最多3字节) while ((unsigned long)p & (sizeof(unsigned) - 1)) { if (*p == '\0') return p - s; p++; } // 阶段2: 每次加载一个 word (4字节), 用 SWAR 检测 const unsigned *wp = (const unsigned *)p; while (1) { unsigned w = *wp++; // SWAR 零字节检测: 若有0字节则 y != 0 unsigned y = (w - 0x01010101u) & ~w & 0x80808080u; if (y != 0) { // 找到包含0字节的 word, 精确定位 p = (const char *)(wp - 1); // 大端: p += nlz(y) >> 3 // 小端: p += ntz(y) >> 3 if (!p[0]) return p - s; if (!p[1]) return p - s + 1; if (!p[2]) return p - s + 2; return p - s + 3; } } } // 性能: ~1字节/周期 (理论极限受内存带宽限制) // 64位版: 每次处理8字节, 使用 0x0101010101010101ULL // AVX2版: 每次处理32字节! (glibc 现代实现) // ARM64: 使用 LDP (load pair) + CMEQ + UMAXV + CLZ
§6-2 查找给定长度的1串 — Find First String of 1-Bits of Length ≥ n
// DISK ALLOCATION — 找第一个足够长的连续空闲块 这个问题来自磁盘块分配器：位图中每个1表示空闲块，需要找到第一个 长度 ≥ n 的连续1串（即连续空闲区域），用于存储新文件。 Warren §6-2 的算法思想精妙： 先消除所有短于 n 的1串（循环 AND 移位），剩下的就是"足够长"的串， 再用 nlz 定位第一个。关键是 x &= (x << 1) 每次能将所有串长度减1， 经过 n-1 次后，只剩下原来长度 ≥ n 的串（已缩短了）。

§6-2 ffstr1 实现 — Warren Figure 6-5
// 查找第一个长度 ≥ n 的1串 // 返回最左1位的位置(0~31), 未找到返回32 int ffstr1(unsigned x, int n) { int i = 1; // 核心: 每次 x &= (x << 1) // 效果: 将所有串长度减1(短串变0,长串缩短) // 重复 n-1 次后, 只剩长度 ≥ n 的串(已缩短) if (n <= 1) goto done; x &= x << 1; // 连续2个1 → 1 if (n <= 2) goto done; // 展开以提高效率(实际用循环或二分) x &= x << 2; // 连续4个1 → 1 (合并步骤) i += 2; // ... done: return nlz(x); // 0~31: 第一个剩余1的位置; 32: 无 } // 改进版: 用二分加速 (Warren Figure 6-6) int ffstr1_fast(unsigned x, int n) { int s = 1; if (n > 16) { x &= x << 16; s += 16; n -= 16; } if (n > 8) { x &= x << 8; s += 8; n -= 8; } if (n > 4) { x &= x << 4; s += 4; n -= 4; } if (n > 2) { x &= x << 2; s += 2; n -= 2; } if (n > 1) { x &= x << 1; } return nlz(x); } // 只需 5 步(log2 32), 每步 2 条指令: ~12条总计
x &= (x << 1) 的作用示例
// 输入: x = 0b 0111 0011 1110 0111
初始 : 0111 0011 1110 0111 (串: 3,4,3 长)
1次AND : 0110 0011 1100 0110 (串: 2,3,2)
2次AND : 0100 0011 1000 0100 (串: 1,2,1)
3次AND : 0000 0011 0000 0000 (串: 1) ← 长度≥4的串
4次AND : 0000 0010 0000 0000 (串: 0,消失)
结论: 3次后只有原长度≥4的串存活,
nlz(result) 给出最左位置
⚡ 核心洞察: AND+左移 = "检查相邻两位都是1"
重复n-1次 = "检查连续n位都是1"
这是串行扫描的位并行加速版本
§6-3 查找最长1串 — Find Longest String of 1-Bits
// HSIEH'S ALGORITHM — 迭代消除最短尾部，计数步数 查找最长1串问题由 Paul Hsieh（2005年发布于 comp.lang.c）给出了优雅的解法： 重复执行 x &= (x << 1)，直到 x 变为 0。 执行的次数恰好等于最长1串的长度！ 原因：每次操作将所有串长度减1，当最长串也变为0时，操作次数 = 最长串长度。 位置可通过记录上一次非零值 oldx，用 nlz 定位。 这是一种"通过退出条件反向推断答案"的巧妙策略，循环次数不超过32次。

§6-3 fmaxstr — 最长1串算法 (Warren Figure 6-7 / Hsieh 2005)
// §6-3 Warren Figure 6-7: 查找最长1串 // Hsieh算法 [comp.lang.c, April 29, 2005] // 原理: 重复 x &= (x << 1) 使所有串长度-1 // 循环次数恰好 = 最长串长度 int fmaxstr(unsigned x) { int k = 0; while (x != 0) { x &= x << 1; // 每次将所有串长度减1 k++; // 计数: 做了多少次 } return k; // k = 最长串长度, x=0最坏32次 } // 变体: 同时记录位置 (Warren 习题1答案) // by Norbert Juffa (cited in Warren answers) int fmaxstr1(unsigned x, int *apos) { int k; unsigned oldx; oldx = 0; for (k = 0; x != 0; k++) { oldx = x; x &= 2 * x; // 等价于 x &= (x << 1) } *apos = nlz(oldx); // oldx 为最后一次非零值 return k; // 起始位 = nlz(oldx) } // 为什么 oldx 的最高1位是起始位? // 因为最长串每轮缩短1位(左端被消掉1位) // 最后非零时, 最高位恰好是原串的起始位 // 性能分析: // 最好 O(1): 只有单个1, 第1次就变0 // 最坏 O(32): 全部为1 (0xFFFFFFFF), 32次 // 平均: 对随机输入约 log2(n) 次 // 每次循环: 2条指令(乘+AND 或 SHL+AND)
§6-4 查找最短1串 — Find Shortest String of 1-Bits
// FMINSTR — AND-SHIFT ELIMINATION APPROACH 查找最短1串问题是 §6-2 问题的逆问题：找到长度最小（但 ≥ 1）的连续1串。 Warren §6-4 的方法同样利用 AND-移位消除法，但方向相反。 核心思路：从 n=1 开始，检查是否存在长度恰好为 n 的1串； 若存在则返回，否则 n++。 检测长度恰好为 n 的串的方法：先找到所有长度 ≥ n 的串（用 ffstr1 的方法）， 再去掉所有长度 ≥ n+1 的串（即两者取差）。 等价于：ffstr(x, n) XOR ffstr(x, n+1) 有差异时，n 就是最短串长度。 更优雅的迭代实现：循环 x &= (x << 1)，每次都能淘汰最短的串，记录步数。

§6-4 fminstr — 两种实现
// §6-4 Warren Figure 6-8: fminstr1 — 查找最短1串 // 方法: 逐步增加 n, 检查是否有长度 ≥ n 的串 // 等价于检查 x &= (x<<1) 后是否消灭了某个串 int fminstr1(unsigned x) { unsigned y; int k = 1; if (x == 0) return 0; while (1) { y = x & (x << 1); // 找长度 ≥ 2 的串 if (y == 0) return k; // 无长度≥2串 → 最短=k k++; x = y; // 继续缩短，检查 k+1 } } // 关键洞察: // x & (x<<1): 保留原串中"两端都是1"的位 // 即每个串从右端被截掉1位 // 原长度=1的串消失; 原长度=2的串变单位 // 当 y==0 时，原 x 的所有串长度都 ≤ k // 由于循环是从1开始的, 退出时 k 即最短串长 // 方法2: 利用 pop(x) 变化来检测 (Warren 习题3) // 每次关闭x中最右边的1串, 观察 pop 减少量 int fminstr_pop(unsigned x) { int p0 = pop(x), p1, minlen = 32; while (x != 0) { // 关闭最右边的1串: x &= x+lowbit(~x) unsigned t = x | (x - 1); // 填充最右0串 x = x & (t + 1); // 清除最右1串 p1 = pop(x); int runlen = p0 - p1; // 该串的长度 if (runlen < minlen) minlen = runlen; p0 = p1; } return minlen; } // 对随机输入平均~10.7次循环(约1/3的32位字 // 最短串为1; ~1%的字最短串为2)

// FROM DISK ALLOCATORS TO DNA SEQUENCE SEARCH 第6章的四个原语看似学术，实则遍布系统软件的每个角落。 strlen 是 C 运行时中调用最频繁的函数之一，其 SWAR 实现每年节省数亿 CPU 周期。 磁盘块分配（ext4/btrfs的 bitmap allocator）用 ffstr1 找连续空闲块。 DNA序列搜索（SIMD BNDM算法）用位并行的1串搜索代替逐字节比较。 这些应用共同验证了：**位级并行**是在 SIMD 硬件普及之前， 最接近"免费午餐"的性能加速手段。
🖥️ strlen / memchr
glibc、musl、Apple libc 的 strlen 全部基于 §6-1 的 SWAR 0字节检测。 64位机每次处理8字节，比逐字节快8倍。 AVX2 版本更可每次处理32字节。

💾 磁盘块分配器
Linux ext4 的块分配器用位图管理空闲块， 用 ffstr1(bitmap_word, n) 找第一个足够大的连续空闲区。 分配 n 个连续块 → 位图中找长度 ≥ n 的1串。

🧬 DNA序列匹配
BNDM（Backward Nondeterministic Dawg Matching）算法 用位并行搜索模式串。将 DNA 碱基编码为位向量， 用 AND 操作同时匹配多个位置，本质是 §6-2 的泛化。
// 位并行 DNA 模式匹配 // A=00 T=01 G=10 C=11 // 4字节 = 16个碱基 D = D & B[text[pos]]; // AND 更新 if (D & accept) found();
🖼️ RLE图像压缩
行程编码（RLE）需要找图像扫描线中最长的连续相同像素串。 二值图像（传真、文档扫描）的 RLE 就是 fmaxstr 的直接应用： 找最长0串和1串交替编码。
// 二值 RLE 编码核心 while (x) { int zeros = ntz(x); // 0串长 x >>= zeros; int ones = ntz(~x); // 1串长 emit(zeros, ones); x >>= ones; }
🔐 哈希表探测
Swiss Table（C++ absl::flat_hash_map）用 SWAR 技术 同时检测16个槽位的元数据字节，判断是否匹配目标 key 的指纹。 本质是 §6-1 的"查找特定字节值"变体，每次比较16个槽。

第6章：SWAR哲学——一个寄存器当四个处理器用

✦ SWAR：比 SIMD 早20年的并行思维

"SIMD Within A Register"（SWAR）这个词在 1998 年才被 Fisher 和 Dietz 正式命名， 但 §6-1 的 0字节检测算法早在 1987 年就被 Alan Mycroft 发现并应用于 strlen。 这告诉我们：真正的好思想往往超前于它的命名。 在没有 MMX/SSE 的年代，程序员用纯整数算术做到了"一次操作四个字节"， 这种精神与 SIMD、GPU、TPU 的向量计算是同源的—— 只是工具不同，洞察相同：数据中的并行性值得被充分利用。

✦ fmaxstr 的"反向计数"哲学——用退出条件反推答案

Hsieh 的最长1串算法是一种罕见的算法设计范式： 不直接计算"答案是多少"，而是通过"迭代到某个终止状态， 用迭代次数反推答案"。这与 Floyd 的环检测、 Gosper 的循环长度算法（§5-4）有异曲同工之妙。 当直接计算很困难时，通过构造一个"自毁"过程来间接量化目标， 是算法设计中的深刻技巧。这类思维在摊还分析、势能函数方法中也有体现。

✦ (x - 0x01010101) & ~x & 0x80808080 — 最优雅的位运算公式之一

这个三运算符的 0字节检测公式堪称位运算史上的经典。 它的正确性建立在对 b=0x80 的精确分析上： 当 b=0x80 时，(b-1)=0x7F，bit7=0，因此不会被误判。 这一"边界案例恰好正确"的巧妙性让每一个理解它的程序员都发出"啊哈！"的惊叹。 Warren 选择用一整页来逐字节分析这个公式的正确性，这种严谨性本身就是一种教育范本。

✦ ffstr1 的"二分加速和-移位"——再次证明二分思维的普适性

§6-2 的二分加速实现（每次移位量对折）将最坏 O(n) 步缩减为 O(log n) 步。 这与 §5-3 的 nlz 二分实现、快速幂、GCD 的二进制算法是同一个二分思维的不同面貌。 Warren 的书里充满了这种"把线性转变为对数"的转化， 而每一次转化都源于对问题结构的深刻理解，而非盲目套用模板。

⚠ 局限性：端序依赖与现代 SIMD 的超越

§6-1 的 SWAR 实现有一个不常被提及的隐患： 大端机和小端机对"第一个0字节位置"的定义不同， 代码中的 nlz(y) >> 3 只在大端机上正确， 小端机需要换成 ntz(y) >> 3。 Warren 在正文中提到了这一点，但很多引用此代码的工程师往往忽略这个细节。 更重要的是：现代 SIMD 指令（SSE4.2 PCMPISTRI、ARM NEON VMOVN） 已经完全超越了 SWAR 的性能上界。 但理解 SWAR 的原理，对于理解这些 SIMD 指令的设计动机至关重要。
